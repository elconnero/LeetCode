class Solution(object):
    def isAdditiveNumber(self, num):
        
        for i in range(1, len(num) - 1):
            for j in range(i + 1, len(num)):
                a = int(num[0:i])
                b = int(num[i:j])
                remaining = num[j:]
                
                while remaining:
                    c = a + b
                    if not remaining.startswith(str(c)):
                        break
                    remaining = remaining[len(str(c)):]
                    a = b
                    b = c
                else:
                    return True
        return False
